# MongoDB with Node.js — Mongoose

## Simple Explanation
**Mongoose** is the most popular Node.js library for working with MongoDB. It provides:
- **Schema** — define the structure of your documents
- **Model** — the interface to communicate with MongoDB
- **Validation** — built-in field validation
- **Middleware** — hooks that run before/after operations
- **Query API** — clean, chainable queries

Think of Mongoose as a **layer between Node.js and MongoDB** that adds structure, safety, and convenience.

---

## Real-World Example
Without Mongoose, you'd talk to MongoDB using raw drivers — lots of boilerplate. With Mongoose:

```js
// Without raw driver complexity:
const user = await User.findOne({ email: "n@n.com" }).select("-password");
// Mongoose handles connection, query, type-casting, and returns a clean object
```

## Code Examples

### 1. Connect to MongoDB
```js
// db.js
const mongoose = require("mongoose");

async function connectDB() {
  try {
    await mongoose.connect(process.env.MONGODB_URI);
    console.log("✅ MongoDB connected");
  } catch (err) {
    console.error("❌ Connection failed:", err.message);
    process.exit(1);
  }
}

// Listen to connection events
mongoose.connection.on("connected", () => console.log("Mongoose connected"));
mongoose.connection.on("error", (err) => console.error("Mongoose error:", err));
mongoose.connection.on("disconnected", () => console.log("Mongoose disconnected"));

module.exports = connectDB;
```

### 2. Define a Schema
```js
const mongoose = require("mongoose");

const productSchema = new mongoose.Schema(
  {
    name: {
      type: String,
      required: [true, "Product name is required"],
      trim: true,
      maxlength: [100, "Name can't exceed 100 characters"]
    },
    price: {
      type: Number,
      required: true,
      min: [0, "Price must be non-negative"]
    },
    category: {
      type: String,
      enum: ["electronics", "clothing", "books", "food"],
      default: "electronics"
    },
    tags: [String],                          // array of strings
    inStock: { type: Boolean, default: true },
    specs: {                                 // nested object
      brand: String,
      model: String,
      warranty: { type: Number, default: 1 }
    },
    sellerId: {
      type: mongoose.Schema.Types.ObjectId,  // reference to users
      ref: "User",
      required: true
    }
  },
  {
    timestamps: true  // auto-adds createdAt and updatedAt fields
  }
);

const Product = mongoose.model("Product", productSchema);
module.exports = Product;
```

// CREATE
const laptop = await Product.create({
  name: "Dell Laptop",

  category: "electronics",
  tags: ["portable", "gaming"],
  sellerId: userId

// READ

// READ with projection + sort + pagination
const products = await Product.find({ inStock: true })
  .select("name price category")     // include only these fields
  .sort({ price: -1 })               // sort by price descending
  .limit(10)
  .skip(20);

// UPDATE
await Product.findByIdAndUpdate(
  "64abc...",
  { $set: { price: 52000, onSale: true } },
  { new: true, runValidators: true }  // return updated doc + validate
);

await Product.updateMany(
  { category: "electronics" },
  { $set: { hasTax: true } }
);

// DELETE
await Product.findByIdAndDelete("64abc...");
await Product.deleteMany({ inStock: false });

### 4. Mongoose Populate — Join Collections
```js
// Order schema references User and Product
const orderSchema = new mongoose.Schema({
  userId: { type: mongoose.Schema.Types.ObjectId, ref: "User" },
  items: [
    {
      productId: { type: mongoose.Schema.Types.ObjectId, ref: "Product" },
      quantity: Number
    }
  ],
  total: Number,
  status: { type: String, default: "pending" }
});

const Order = mongoose.model("Order", orderSchema);

// Fetch order with full user and product data
const order = await Order.findById(orderId)
  .populate("userId", "name email")           // only name and email from User
  .populate("items.productId", "name price"); // name and price from Product

console.log(order.userId.name);              // "Nishant"
console.log(order.items[0].productId.name); // "Dell Laptop"
```

### 5. Mongoose Middleware (Hooks)
```js
const bcrypt = require("bcryptjs");
const userSchema = new mongoose.Schema({ ... });

// Pre-save hook — hash password before saving
userSchema.pre("save", async function (next) {
  if (!this.isModified("password")) return next(); // skip if unchanged
  this.password = await bcrypt.hash(this.password, 10);
  next();
});

// Post-save hook — log after saving
userSchema.post("save", function (doc) {
  console.log(`User saved: ${doc.email}`);
});

// Pre-find hook — always exclude __v
userSchema.pre(/^find/, function (next) {
  this.select("-__v");
  next();
});
```

### 6. Mongoose Virtual Fields
```js
// Computed field not stored in DB
userSchema.virtual("fullName").get(function () {
  return `${this.firstName} ${this.lastName}`;
});

// Include virtuals in JSON output
userSchema.set("toJSON", { virtuals: true });
userSchema.set("toObject", { virtuals: true });

const user = await User.findById(id);
console.log(user.fullName); // "Nishant Kumar"
```

### 7. Mongoose Instance & Static Methods
```js
// Instance method — on a specific document
userSchema.methods.comparePassword = async function (inputPassword) {
  return bcrypt.compare(inputPassword, this.password);
};

// Static method — on the Model itself
userSchema.statics.findByEmail = function (email) {
  return this.findOne({ email });
};

// Usage
const user = await User.findByEmail("n@n.com");
const isMatch = await user.comparePassword("mypassword");
```

## Mongoose API Cheat Sheet

| Method | Description |
|--------|-------------|
| `Model.create(data)` | Create and save a document |
| `Model.find(filter)` | Find all matching documents |
| `Model.findOne(filter)` | Find first matching document |
| `Model.findById(id)` | Find by `_id` |
| `Model.findByIdAndUpdate(id, update, opts)` | Find and update by `_id` |
| `Model.findByIdAndDelete(id)` | Find and delete by `_id` |
| `Model.updateMany(filter, update)` | Update all matching |
| `Model.deleteMany(filter)` | Delete all matching |
| `.populate(field)` | Replace ObjectId with actual document |
| `.select("field1 field2")` | Include/exclude fields |
| `.sort({ field: 1/-1 })` | Sort ascending/descending |
| `.limit(n)` | Return max n documents |
| `.skip(n)` | Skip n documents |
| `.countDocuments(filter)` | Count matching documents |
| `.lean()` | Return plain JS object (faster, no Mongoose overhead) |

## Interview Questions

- **Q: What is Mongoose?**
  A: A Node.js ODM (Object Document Mapper) for MongoDB that provides schema definition, type casting, validation, middleware, and a clean query API.

- **Q: What is the difference between a Schema and a Model in Mongoose?**
  A: A Schema defines the structure, fields, and validation rules. A Model is a constructor compiled from the Schema — used to interact with the MongoDB collection.

- **Q: What is `populate()` in Mongoose?**
  A: A method that replaces ObjectId reference fields with actual document data from the referenced collection — similar to a SQL JOIN.

- **Q: What is a Mongoose hook (middleware)?**
  A: Functions that run before (`pre`) or after (`post`) operations like `save`, `find`, or `remove`. Used for hashing passwords, logging, or auto-populating fields.

- **Q: What does `.lean()` do in Mongoose?**
  A: Returns plain JavaScript objects instead of Mongoose Document instances — no methods or virtuals, but much faster for read-only queries.

- **Q: What is `timestamps: true` in Mongoose?**
  A: Automatically adds `createdAt` and `updatedAt` fields to every document, managed by Mongoose.

- **Q: What is `runValidators: true` in update operations?**
  A: By default, Mongoose only runs validators on `save()`. `runValidators: true` option makes validators run on `update` operations too.
